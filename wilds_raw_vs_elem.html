#Raw damage
def raw_dmg(base_raw, sharpness='G', motion_value=100, hitzone=100, 
            flat_raw_mod=0, base_raw_mod=1, total_raw_mod=1):
    sharp_mod = raw_sharp_mod[str(sharpness).capitalize()]
    return (base_raw*base_raw_mod+flat_raw_mod)*total_raw_mod*sharp_mod*(motion_value/100)*(hitzone/100)

#Effective raw damage (linearly averages affinity)
def eff_raw_dmg(base_raw, sharpness='G', motion_value=100, hitzone=100, 
                affinity=0, crit_boost=0,
                flat_raw_mod=0, base_raw_mod=1, total_raw_mod=1):
    return raw_dmg(base_raw,sharpness,motion_value,hitzone,flat_raw_mod,base_raw_mod,total_raw_mod)*(1+(affinity*(crit_boost*3+25))/100^2)

#Elemental damage
def ele_dmg(base_element, sharpness='G', hitzone=100, 
            flat_ele_mod=0, base_ele_mod=1, total_ele_mod=1):
    sharp_mod = ele_sharp_mod[str(sharpness).capitalize()]
    return (base_element*base_ele_mod+flat_ele_mod)*sharp_mod*(hitzone/100)*total_ele_mod

#Effective elemental damage
def eff_ele_dmg(base_element, sharpness='G', hitzone=100, 
                affinity=100, crit_element=0, 
                flat_ele_mod=0, base_ele_mod=1, total_ele_mod=1):
    return ele_dmg(base_element,sharpness,hitzone,flat_ele_mod,base_ele_mod,total_ele_mod)*(1+(affinity*crit_element*7)/100^2)

#Ratio of base raw to base element
def raw_vs_ele(raw_hz, ele_hz, sharpness='G', motion_value=100, 
               affinity=0, crit_boost=0, crit_element=0, 
               flat_raw_mod=0, base_raw_mod=1, total_raw_mod=1, 
               flat_ele_mod=0, base_ele_mod=1, total_ele_mod=1):
    efr = eff_raw_dmg(base_raw,sharpness,motion_value,raw_hz,affinity,crit_boost,
                                  0,base_raw_mod,total_raw_mod)
    efe = eff_ele_dmg(base_element,sharpness,ele_hz,affinity,crit_element,
                                  0,base_ele_mod,total_ele_mod)
    if ele_hz==0: return 10
    else: ratio = (solve(efr-efe,base_element)[0]/base_raw).rhs()
    return ratio

  • Note that elemental values in Wilds are bloated by 10, e.g., 200 base elmenent is 20 "true" element.
  • Use Total Elemental Modifier to enter whatever appropriate modifiers are coming from skills.
  • Use eMV to account for the total elemental modifier coming from a sequence of moves being analyzed. This parameter is only here to simplify calculation of EFE vs ERE. E.g, if you analyze three moves with elemental modifiers of 1, 1, and 0.7, then you input 2.7 for eMV.
